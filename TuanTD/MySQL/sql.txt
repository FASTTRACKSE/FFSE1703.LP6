1.SELECT students.name,  YEAR(CURDATE()) - YEAR( students.birth_day) AS Tuoi, thanh_pho.ten_thanh_pho, quoc_tich.ten_quoc_tich AS ten_nuoc, nation.ma_quoc_tich FROM((( students LEFT JOIN thanh_pho ON students.ma_thanh_pho = thanh_pho.ma_thanh_pho) LEFT JOIN quoc_tich ON students.ma_nuoc = quoc_tich.ma_quoc_tich) LEFT JOIN quoc_tich nation ON students.ma_quoc_tich = nation.ma_quoc_tich) WHERE students.ma_nuoc = 'VN'

2.SELECT students.name, YEAR(CURDATE()) - YEAR( students.birth_day) AS 'Tuoi', thanh_pho.ten_thanh_pho , quoc_tich.ma_quoc_tich AS 'Nuoc', quoc_tich.ten_quoc_tich FROM (((students INNER JOIN thanh_pho ON students.ma_thanh_pho = thanh_pho.ma_thanh_pho) LEFT JOIN quoc_tich ON students.ma_quoc_tich = quoc_tich.ma_quoc_tich) LEFT JOIN quoc_tich  quoc_tichqt ON students.ma_nuoc = quoc_tichqt.ma_quoc_tich )
WHERE students.ma_quoc_tich = 'VN' AND students.ma_nuoc= 'VN'

3.SELECT students.name, YEAR(CURDATE()) - YEAR( students.birth_day) AS Tuoi, thanh_pho.ten_thanh_pho, quoc_tich.ten_quoc_tich AS ten_nuoc, nation.ma_quoc_tich, lophocchinhquy.ten_lop, diem_hoc_chinh_quy.LP1,diem_hoc_chinh_quy.LP2,diem_hoc_chinh_quy.LP3,diem_hoc_chinh_quy.LP4,diem_hoc_chinh_quy.LP5,diem_hoc_chinh_quy.LP6 FROM (((((students LEFT JOIN thanh_pho ON students.ma_thanh_pho = thanh_pho.ma_thanh_pho) LEFT JOIN quoc_tich ON students.ma_nuoc = quoc_tich.ma_quoc_tich) LEFT JOIN quoc_tich nation ON students.ma_quoc_tich = nation.ma_quoc_tich) LEFT JOIN diem_hoc_chinh_quy ON students.masv = diem_hoc_chinh_quy.masv) LEFT JOIN lophocchinhquy ON students.ma_lop = lophocchinhquy.ten_lop)

4*.SELECT COUNT(students.ma_lop) AS tong_sinh_vien, lophocchinhquy.ma_lop, MAX((diem_hoc_chinh_quy.LP1+diem_hoc_chinh_quy.LP2+diem_hoc_chinh_quy.LP3+diem_hoc_chinh_quy.LP4+diem_hoc_chinh_quy.LP5)/ 5) AS diem_TB_Cao_Nhat, MIN((diem_hoc_chinh_quy.LP1+diem_hoc_chinh_quy.LP2+diem_hoc_chinh_quy.LP3+diem_hoc_chinh_quy.LP4+diem_hoc_chinh_quy.LP5)/ 5) AS diem_TB_Thap_Nhat FROM ((students LEFT JOIN lophocchinhquy ON students.ma_lop = lophocchinhquy.ma_lop) LEFT JOIN diem_hoc_chinh_quy ON students.masv = diem_hoc_chinh_quy.masv) GROUP BY lophocchinhquy.ma_lop

5*.SELECT students.masv, YEAR(CURDATE()) - YEAR(students.birth_day) AS 'Tuoi' , students.name, BS1.diem AS 'BS1', BS2.diem AS 'BS2', BS3.diem AS 'BS3', AVG(diem_hoc_bo_sung.diem) AS 'DTB' FROM students LEFT JOIN diem_hoc_bo_sung BS1 ON BS1.ma_mon_hoc = 'BS1' AND BS1.masv = students.masv LEFT JOIN diem_hoc_bo_sung BS2 ON BS2.ma_mon_hoc = 'BS2' AND BS2.masv = students.masv LEFT JOIN diem_hoc_bo_sung BS3 ON BS3.ma_mon_hoc = 'BS3' AND BS3.masv = students.masv LEFT JOIN diem_hoc_bo_sung ON diem_hoc_bo_sung.masv = students.masv GROUP BY students.masv
